* Case shift routines

UPSHIFT  CMP   #"z"+1
         BGE   :NO
         CMP   #"a"
         BLT   :NO
         AND   #%11011111 convert to uppercase
:NO      RTS

DNSHIFT  CMP   #"A"
         BLT   :NO
         CMP   #"Z"+1
         BGE   :NO
         ORA   #%00100000 convert to lowercase
:NO      RTS
